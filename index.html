Problema 1

Realiza la convolución de las siguientes señales.
clear all
figure
subplot(1,2,1)
syms t
x=piecewise(t<0,0,0<t<1,-t+1,1<t<2,t-1,t>2,0);
fplot(x,[-0.25,2.25],'r')
grid on
title('x(t)')
axis([-1 3.6 -.25 1.5]);
subplot(1,2,2)
h=piecewise(t<0,0,0<t<1,1,t>1,0);
fplot(h,[-0.25,1.25],'r')
grid on
title('h(t)')
axis([-1 1.6 -.25 1.5]);
syms tau
assume(t, 'real')
assume(tau, 'real')
f(t) = (-t+1)*(heaviside(t)-heaviside(t-1))+(t-1)*(heaviside(t-1)-heaviside(t-2));
h(t) = (heaviside(t)-heaviside(t-1));
g(t) = int(f(tau)*h(t-tau),tau,-inf,inf);
subplot(2,1,1, 'align')
fplot(subs(g(t),tau,2))
grid on
title('x(t)*h(t)')
axis([-1 4 -1 2]);
x = @(t) ((-t+1).*(t>=0 & t<1) + (t-1).*(t>=1 & t<2));
h = @(t) 1*(t>=0 & t<1);
convconm(x,h)

x = @(t) ((t).*(t>=0 & t<1) + (1).*(t>=1 & t<2));

convconm(x,h)
